- ทำการรับ Plain Text (ข้อมูลที่สามารถอ่านได้)
- ทำการรับ Key (จำนวนในการเลื่อนตำแหน่ง)

- ทำการรับ Cipher Text (ข้อมูลที่ถูกเข้ารหัสเอาไว้ไม่สามารถอ่านได้)
- ทำการรับ Key


 
    import java.util.Scanner;

    public class Caesar {

        public static void main(String[] args) {
        
            String plainText;
            int key;
            Scanner in = new Scanner(System.in);

            System.out.print("input plain text : ");
            plainText = in.next().toUpperCase();

            System.out.print("input Key : ");
            key = in.nextInt();
          
            System.out.println( "encript : " + encript(plainText,key));
        
        }
      
        public static String encript(String plainText,int key) {
            String cipherText = "";
            key = key % 26;
            for (int i = 0; i < plainText.length(); i++) {
                char temp = (char) (plainText.charAt(i) + key);
                if (temp > 90) {
                    cipherText += (char) ('A' + (temp - 91));
                }else{
                    cipherText += temp;
                }
            }
            return cipherText;
        }
      
    }
