                if (temp > 90) {
                    cipherText += (char) ('A' + (temp - 91));

                    cipherText += temp;
                }

       public static String decrypt(String cipherText,int key) {
        String plainText = "";
        key = key % 26;
        for (int i = 0; i < cipherText.length(); i++) {
            char temp = (char) (cipherText.charAt(i) - key);
            if (temp < 65) {
                plainText += (char) ('Z' - (64 - temp));
            }else{
                plainText += temp;
            }
        }
        return plainText;
      }
