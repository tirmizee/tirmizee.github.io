และถ้าหากว่าข้อความดังกล่าวถูกแย่งชิงระหว่างทางโดยข้าศึก และข้าศึกไม่รู้กุญแจในการถอดรหัสก็ไม่สามารถรู้ได้ว่าข้อความดังกล่าวคืออะไร

        public static void main(String[] args) {
        
            String plainText;                               //ประกาศตัวแปรเป็นชนิดข้อมูลเก็บข้อความ

            Scanner in = new Scanner(System.in);            //ประกาศตัวแปรที่ใช้รับค่าที่แสกนจากแป้นพิมพ์ 

            System.out.print("input plain text : ");        //แสดงข้อความ "input plain text : "
            plainText = in.next().toUpperCase();            //ทำการรับข้อความจากแป้นพิมพ์พร้อมกับแปลงเป็นตัวอักษรพิมพ์ใหญ่เก็บไว้ในตัวแปร plainText

            System.out.print("input Key : ");               //แสดงข้อความ "input Key : "
            key = in.nextInt();                             //ทำการรับเลขจำนวนเต็มจากแป้นพิมพ์เก็บไว้ในตัวแปร key
          
            System.out.println( "encript : " + encript(plainText,key));        //แสดงข้อความที่ถูก encript
        
        }
      
        public static String encript(String plainText,int key) {
            String cipherText = "";         
            char temp;
            key = key % 26;                 
            for (int i = 0 , length = plainText.length(); i < length; i++) {         //ทำการวนลูปตามจำนวนตัวอักษรใน plainText
                temp = (char) (plainText.charAt(i) + key);          //ทำการเลือนตัวอักษรตามจำนวนของ key 
                if (temp > 'Z') {                    //ทำการเชคเงื่อนไขว่าตัวอักษรที่ถูกเลื่อนมีค่ามากกว่า 'Z' หรือเปล่า
                    //ถ้า temp > 'Z' ทำการเลื่อนตัวอักษรตามจำนวนที่เกิน 'Z' โดยเริ่มต้นเลื่อนไปยังตัวอักษร 'A' อีกครั้ง
                    temp = (char) (temp - 26);         
                }
                cipherText += temp; 
            }
            return cipherText;
        }
      
    }

       public static String decrypt(String cipherText,int key) {
        String plainText = "";
        key = key % 26;
        for (int i = 0; i < cipherText.length(); i++) {
            char temp = (char) (cipherText.charAt(i) - key);
            plainText += temp;
            if (temp < 'A') {
                plainText += (char) ('Z' - (64 - temp));
            }
        }
        return plainText;
      }
