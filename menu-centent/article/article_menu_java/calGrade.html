วิธีนี้เป็นการตรวจสอบเงื่อนไขเป็นชุดหลายทางเลือก หากเข้าเงื่อนไขใด โปรแกรมจะออกจากชุดของ if นั้นทันที 

import java.util.Scanner;
public class Grade {

  public static void main(String[] args) {

    int score;                              //ประกาศตัวแปรเป็นชนิดข้อมูลเก็บจำนวนเต็ม 
    String grade;                           //ประกาศตัวแปรเป็นชนิดข้อมูลเก็บข้อความ 
    Scanner in = new Scanner(System.in);    //ประกาศตัวแปรที่ใช้รับค่าที่แสกนจากแป้นพิมพ์ 
    System.out.print("input score : ");     //แสดงข้อความ "input score" 
    score = in.nextInt();                   //ทำการรับค่าจำนวนเต็มจากแป้นพิมพ์เก็บไว้ในตัวแปร score 
    
    if (score < 50) {                       //ทำการเชคเงื่อนไขว่า score < 50 จริงไหม 
      grade = "F";                          //ถ้าเงื่อนไขด้านบนเป็นจริง กำหนดให้ grade มีค่าเท่ากับ "F" 
    }else if(score >= 50 && score < 55){    //บลาๆๆ... 
      grade = "D";
    }else if(score >= 55 && score < 60){
      grade = "D+";
    }else if(score >= 60 && score < 65){
      grade = "C";
    }else if(score >= 65 && score < 70){
      grade = "C+";
    }else if(score >= 70 && score < 75){
      grade = "B";
    }else if(score >= 75 && score < 80){
      grade = "B+";
    }else {
      grade = "A";
    }

    System.out.println("Grade : " + grade);
       
  }
}

...แต่เอ๊ะ !!! ดูเหมือนว่าการเชคเงื่อนไขมันทำงานซ้ำซ้อนกันอยู่ ลองมาดูโค้ดด่านล่างกันครับ

  1.  if( score < 50) {         
  2.    grade = "F";                          
  3.  }else if(score >= 50 && score < 55){     
  4.    grade = "D";
  5.  }
ถ้าเงื่อนไขบรรทัดที่ 1 เป็นเท็จ แสดงว่า score ต้องมีค่า >= 50 แล้วโปรแกรมจะไปทำบรรทัดที่ 3
แต่ลองสังเกตบรรทัดที่ 3 ดู พบว่ามีการเชค score >= 50 ซ้ำอีกรอบ
ซึ่งเราไม่จำเป็นต้องเชคอีก เพราะถ้าหากมันเข้าเงื่อนไขบรรทัดนี้แสดงว่า score ต้องมีค่า >= 50 อยู่แล้ว

สมมุติว่า ต้องการตรวจสอบ score มีค่ามากกว่า

import java.util.Scanner;
public class Grade {

  public static void main(String[] args) {

    int score;                              //ประกาศตัวแปรเป็นชนิดข้อมูลเก็บจำนวนเต็ม 
    String grade;                           //ประกาศตัวแปรเป็นชนิดข้อมูลเก็บข้อความ 
    Scanner in = new Scanner(System.in);    //ประกาศตัวแปรที่ใช้รับค่าที่แสกนจากแป้นพิมพ์ 
    System.out.print("input score : ");     //แสดงข้อความ "input score" 
    score = in.nextInt();                   //ทำการรับค่าจำนวนเต็มจากแป้นพิมพ์เก็บไว้ในตัวแปร score 
    
      switch (score/5) {
        case 20 :
        case 19 :
        case 18 :
        case 17 :
        case 16 : grade = "A";              //หาก score มีค่า 20,19,18,17,16 กำหนดให้ grade มีค่าเท่ากับ "A" 
                  break;                    //หยุดการทำงานใน switch case เพื่อไม่ให้โปรแกรมทำคำสั่งถัดไป 
        case 15 : grade = "B+";             //บลาๆๆ... 
                  break;
        case 14 : grade = "B";
                  break;
        case 13 : grade = "C+";
                  break;
        case 12 : grade = "C";
                  break;
        case 11 : grade = "D+";
                  break;
        case 10 : grade = "D";
                  break;  
        default:  grade = "F";            //หากไม่เข้าเงื่อนไขใดๆ กำหนดให้ grade มีค่าเท่ากับ "F" 
                  break;
    }

    System.out.println("Grade : " + grade);
       
  }
}
