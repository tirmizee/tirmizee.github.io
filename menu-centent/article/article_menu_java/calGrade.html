ถ้าคะแนนน้อยกว่า 50 ได้เกรด F
ถ้าคะแนนตั้งแต่ 50-54 ได้เกรด D
ถ้าคะแนนตั้งแต่ 55-59 ได้เกรด D+
ถ้าคะแนนตั้งแต่ 60-64 ได้เกรด C
ถ้าคะแนนตั้งแต่ 65-69 ได้เกรด C+
ถ้าคะแนนตั้งแต่ 70-74 ได้เกรด B
ถ้าคะแนนตั้งแต่ 75-79 ได้เกรด B+
ถ้าคะแนนตั้งแต่ 80 ขึ้นไป ได้เกรด A

วิธีนี้เป็นการตรวจสอบเงื่อนไขเป็นชุด 

import java.util.Scanner;
public class Grade {

  public static void main(String[] args) {

    int score;                              //ประกาศตัวแปรเป็นชนิดข้อมูลเก็บจำนวนเต็ม 
    String grade;                           //ประกาศตัวแปรเป็นชนิดข้อมูลเก็บข้อความ 
    Scanner in = new Scanner(System.in);    //ประกาศตัวแปรที่ใช้รับค่าที่แสกนจากแป้นพิมพ์ 
    System.out.print("input score : ");     //แสดงข้อความ "input score" 
    score = in.nextInt();                   //ทำการรับค่าจำนวนเต็มจากแป้นพิมพ์ 
    
    if (score < 50) {                       //ทำการเชคเงื่อนไขว่า score < 50 จริงไหม 
      grade = "F";                          //ถ้าเงื่อนไขด้านบนเป็นจริง กำหนดให้ grade มีค่าเท่ากับ "F" 
    }else if(score >= 50 && score < 55){    //บลาๆๆ... 
      grade = "D";
    }else if(score >= 55 && score < 60){
      grade = "D+";
    }else if(score >= 60 && score < 65){
      grade = "C";
    }else if(score >= 65 && score < 70){
      grade = "C+";
    }else if(score >= 70 && score < 75){
      grade = "B";
    }else if(score >= 75 && score < 80){
      grade = "B+";
    }else {
      grade = "A";
    }

    System.out.println("Grade : " + grade);
       
  }
}

แต่ลองสังเกตบรรทัดที่ 3 ดู พบว่ามีการเชค score >= 50 ซ้ำอีกรอบ

import java.util.Scanner;
public class Grade {

  public static void main(String[] args) {

    int score;                              //ประกาศตัวแปรเป็นชนิดข้อมูลเก็บจำนวนเต็ม 
    String grade;                           //ประกาศตัวแปรเป็นชนิดข้อมูลเก็บข้อความ 
    Scanner in = new Scanner(System.in);    //ประกาศตัวแปรที่ใช้รับค่าที่แสกนจากแป้นพิมพ์ 
    System.out.print("input score : ");     //แสดงข้อความ "input score" 
    score = in.nextInt();                   //ทำการรับค่าจำนวนเต็มจากแป้นพิมพ์เก็บไว้ในตัวแปร score 
    
      switch (score/5) {
        case 20 :
        case 19 :
        case 18 :
        case 17 :
        case 16 : grade = "A";
                  break;
        case 15 : grade = "B+";
                  break;
        case 14 : grade = "B";
                  break;
        case 13 : grade = "C+";
                  break;
        case 12 : grade = "C";
                  break;
        case 11 : grade = "D+";
                  break;
        case 10 : grade = "D";
                  break;  
        default:  grade = "F";
                  break;
    }

    System.out.println("Grade : " + grade);
       
  }
}
ผู้อ่านที่กำลังเริ่มต้นศึกษาอาจจะตั้งคำถามว่าทำไมเงื่อนไขใน switch case ต้องทำการเชค A ก่อน
แล้วถ้าเริ่มต้นจากเชค F ก่อนมันไม่เหมือนกันเหรอ งั้นเรามาดูเหตผลว่าทำไมต้องทำการเชค F ก่อนจากโค้ดด่านล่าง
